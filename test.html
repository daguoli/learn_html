npm_install() {
  cd $BASEDIR
  if [ -n "$SKIP_NPM" ]; then
    echo "[chair-shell:build] skip tnpm install"
    return
  fi
  echo "[chair-shell:build] tnpm install to $BASEDIR/node_modules"
  rm -rf node_modules
  npm install tnpm $NPM_OPT || exit $?
  tnpm install --production || exit $?
}
